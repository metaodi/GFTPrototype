% command to import a figure
\newcommand{\fig}[5]{
  \begin{figure}[h]
    \begin{center}
      \includegraphics[width=#4cm]{#1}
    \end{center}
    \stepcounter{myfigure}
    \caption[#5]{#3}
    \label{#2}
  \end{figure}
}

\newcommand{\figtable}[4]{

			  #3
      }
    \end{center}
    \stepcounter{myfigure}
    \caption[#4]{#2}
    \label{#1}
  \end{figure}
}

\newcommand{\tab}[4]{
  \begin{table}[h]
    \begin{center}
      {  
			  \footnotesize
  			\sffamily

			  #3  			
      }
    \end{center}
    \stepcounter{mytable}
    \caption[#4]{#2}
    \label{#1}
  \end{table}
}
